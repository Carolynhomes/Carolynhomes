```python
def solution2():
    t = int(input())
    
    # 初始化字典来管理招募人数和最大值
    counts = {i: 0 for i in range(1, 6)}
    max_counts = {i: 0 for i in range(1, 6)}

    for _ in range(t):
        number_list = list(map(int, input().split()))
        
        for i in range(1, 6):
            if i in number_list:
                counts[i] += 1  # 如果当前数字在该行，连续计数加一
            else:
                # 更新最大值并重置连续计数
                max_counts[i] = max(max_counts[i], counts[i])
                counts[i] = 0

    # 检查最后一次连续计数
    for i in range(1, 6):
        max_counts[i] = max(max_counts[i], counts[i])

    # 打印结果
    print(" ".join(str(max_counts[i]) for i in range(1, 6)))


solution2()
```
